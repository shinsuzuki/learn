import { useState } from 'react';

function CounterApp() {
    // ①useStateを使ってカウントのステートを定義
    const [count, setCount] = useState(0);

    const handleClick = () => {
        // ②ボタンがクリックされたときにカウントを1増やす
        setCount(count + 1);
    };

    return (
        <div>
            {/* ③カウントの値を表示 */}
            <p>現在のカウント: {count}</p> 
            {/* ④クリックイベント */}
            <button onClick={handleClick}>+1する</button>
        </div>
    );
}

export default CounterApp;
